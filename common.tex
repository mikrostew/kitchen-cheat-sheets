% commands

% page title
\newcommand{\pagetitle}[1]%
{%
  {\huge\scshape{#1}}%
}%

% section headers - small-caps title
\newcommand{\sectionheader}[1]%
{%
  \bigskip\textsc{#1}\strut%
}%

% ratio table layout
\newcommand{\ratiotable}[1]%
{%
  \rowcolors{1}{green!20}{white}%
  \outlinetabularx{ p{4cm} X c X c X c X }{#1}%
}%

% tabular environment, outlined with rounded corners
\newcommand{\outlinetabular}[2]%
{%
  \tableoutline{%
    \begin{tabular}{ #1 }%
      #2%
    \end{tabular}%
  }%\tableoutline
}%

% full-width tabularx environment, outlined with rounded corners
\newcommand{\outlinetabularx}[2]%
{%
  \tableoutline{%
    \begin{tabularx}{\dimexpr\textwidth-2\pgflinewidth\relax}{ #1 }%
      #2%
    \end{tabularx}%
  }%\tableoutline
}%

\newcommand{\tableoutline}[1]%
{%
  \addvbuffer[4pt 0pt]{%
    % rounded corners
    % (see http://tex.stackexchange.com/q/184060)
    \noindent\begin{tikzpicture}[line width=.1mm,rounded corners=.5em]%
      \node(thetable) [clip,inner sep=.5\pgflinewidth] {%
        \addstackgap[\pgflinewidth]{%
          #1%
        }%
      };
      \draw ([xshift=.5*\pgflinewidth,yshift=-.5*\pgflinewidth]thetable.north west)
        rectangle ([xshift=-.5*\pgflinewidth,yshift=.5*\pgflinewidth]thetable.south east);
    \end{tikzpicture}%
  }%\addvbuffer
}%

% for notes under a table
\newcommand{\note}[1]%
{%
  {\small\begin{addmargin}{1em} #1 \end{addmargin}}%
}%
